git status #Tells us which files to push up

git add -A Adds all the files to be saves

git commit -m "Description of what changes" #Describes how the files were changed

git push origin master #Saves the files to git

git checkout -b [branch_name] #Creates a new branch

git branch # List all the branches and tells us which branch we are currently on

git checkout [branch_name] # Switch between branches

git pull origin [branch_name] # Pull the latest changes from git
